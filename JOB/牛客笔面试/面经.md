#### atoi (ascii to int )

```c
int isempty(char* str)
{
    if (str == ' ' ||
        str == '\t' ||
        str == '\n' ||
        str == '\f' ||
        str == '\v' ||
        str == '\r')
    {
        return 1;
    }
    return 0;
}


int isnumber(char* str)
{
    if (str >= '0' && str <= '9')
    {
        return 1;
    }
    return 0;
}

int  my_atoi(char* str)
{
    assert(str != NULL);
    if (*str == '\0')//空字符串问题
	{
		return 0;
    }
    
     while (isempty(*str)==1)//判断是否为空
    {
        str++;
    }
    
    int flg = 1;
	if (*str == '+')
	{
		flg = 1;
		str++;
	}//注意这里要用else if，不能用else
	else if(*str == '-')
	{
		flg = -1;
		str++;
	}
    long long n = 0;
   
    while (isnumber(*str) == 1)//判断是否为数字
    {
        if (isdigit(*str))
		{//是数字字符==>返回值不为0
			n = n * 10 + (*str - '0') * flg;
			if (n > INT_MAX)
			{
				n = INT_MAX;
			}
			if (n < INT_MIN)
			{
				n = INT_MIN;
			}
		}
		else//不是数字字符
		{
			return (int)n;
		}
		str++;
    }
    if (*str == '\0')
	{
		status = VAILD;//走到‘\0’就是合法转换
	}
	return (int)n;
}
```
